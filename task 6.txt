#include <iostream>
using namespace std;

int main() {
    int ay;
    cout << "Ay nomresini daxil et: ";
    cin >> ay;

    if (ay == 12 || ay == 1 || ay == 2) {
        cout << "Qis";
    }
    else if (ay >= 3 && ay <= 5) {
        cout << "Yaz";
    }
    else if (ay >= 6 && ay <= 8) {
        cout << "Yay";
    }
    else if (ay >= 9 && ay <= 11) {
        cout << "Payiz";
    }
    else {
        cout << "Duzgun ay nomresi daxil et!";
    }

    return 0;
}
